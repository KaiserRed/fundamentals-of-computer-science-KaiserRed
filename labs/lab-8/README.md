```
#include <stdio.h>
#include <stdlib.h>

    int a, b,d;

    printf("%d\n", lcm(a,b));
    return 0;        
}   
int gcd(int a, int b) {
    while (b != 0){
        int c = a % b;
        a = b;
        b = c;
    }
    return abs(a); 
}
int lcm(int a,int b){
    return abs(a / gcd(a, b) * b);
}
```
